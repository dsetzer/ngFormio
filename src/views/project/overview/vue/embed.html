Vue.JS

Creating a Vue component

Be sure you have added the formiojs library to your application and imported it in the app declaration.

Embed within a template

To embed a form within your Vue application, lets create a component call FormVue, first you have to import FormioForm from the formiojs library.


                        // FormVue.vue
                        <template>
                           <div id="formio"> </div>
                        </template>
                        <script>
                          import FormioForm from 'formiojs/form';
                          export default {
                             name: 'formio-vue',
                              props: ['src'],
                             mounted () {
                              let self = this;
                              let form = new FormioForm(document.getElementById('formio'));
                              form.src = self.src;
                              form.on('submit', (submission) => {
                               console.log('The form was just submitted!!!');
                             });
                              form.on('error', (errors) => {
                                  console.log('We have errors!');
                             })
                             }
                          }
                        </script>

                        ---
                        // Then we can use our component in our App.vue.
                        <template>
                          <div id="app">
                          <formio-vue src="'{{ projectUrl }}/email'"></formio-vue>
                        </div>
                        </template>
                        <script>
                          import FormioVue from './components/Formio.vue'
                          export default {
                            name: 'app',
                            components: {
                            FormioVue
                          }
                          }
                        </script>
