Add aurelia-formio plugin to your application

export function configure(aurelia: Aurelia) {
  aurelia.use
    .standardConfiguration()
    .plugin(PLATFORM.moduleName('aurelia-formio'));

Add formio component to a template

Place the following where you would like to show this form.

<template>
  <formio src="https://examples.form.io/example" submission.bind="submission"></formio>
</template>

Add properties to the corresponding class

export class MyClass {
  private submission: any = {
    data: {
        firstName: 'Joe',
        lastName: 'Smith'
    }
  };
}
