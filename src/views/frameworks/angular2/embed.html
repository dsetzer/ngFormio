Install angular-formio

Install the angular-formio module.

npm install --save angular-formio

Import the FormioModule

app.module.ts 
import { NgModule } from '@angular/core';
import { FormioModule } from 'angular-formio';
import { BrowserModule } from '@angular/platform-browser';
import { MainComponent } from './main.component.ts';

@NgModule({
   imports: [
      BrowserModule,
      FormioModule
   ],
   declarations: [
      MainComponent
   ],
   bootstrap: [
      MainComponent
   ]
})
export class AppModule {}

Embed the form

Include the <formio> component within MainComponent

main.component.ts 
import { Component } from '@angular/core';
@Component({
     template: '<formio src="{{ projectUrl }}/{{ form.path }}"></formio>'
})
export class MainComponent {}
